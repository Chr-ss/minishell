15int return_quoted_length(char *line, char quote)
16{
17 int quoted_len;
18 int i;
19
20 quoted_len = 0;
21 while (line[quoted_len] && line[quoted_len + 1] && line[quoted_len + 1] != quote)
22 {
23 line[quoted_len] = line[quoted_len + 1];
24 quoted_len++;
25 }
26 i = quoted_len;
27 while (line[i + 2])
28 {
29 line[i] = line[i + 2];
30 i++;
31 }
32 line[i] = '\0';
33 line[i + 1] = '\0';
34 return (quoted_len);
35}

37t_token tokenizer(char *line)
38{
39 t_token token;
40 char *temp;
41
42 token.start = line;
43 temp = token.start;
44 token.length = 1;
45 if (!*token.start)
46 {
47 token.type = TOKEN_EOF;
48 write(1, "EOF, ", 5);
49 write(1, &token.start[0], token.length);
50 write(1, "$\n", 2);
51 }
52 else if (!ft_strncmp(token.start, "|", 1))
53 {
54 token.type = TOKEN_PIPE;
55 write(1, "PIPE, ", 6);
56 write(1, &token.start[0], token.length);
57 write(1, "$\n", 2);
58 }
59 else if (!ft_strncmp(token.start, "<<", 2))
60 {
61 token.type = TOKEN_HEREDOC;
62 token.length = 2;
63 write(1, "HEREDOC, ", 9);
64 write(1, &token.start[0], token.length);
65 write(1, "$\n", 2);
66 }
67 else if (!ft_strncmp(token.start, "<", 1))
68 {
69 token.type = TOKEN_REIN;
70 write(1, "REIN, ", 6);
71 write(1, &token.start[0], token.length);
72 write(1, "$\n", 2);
73 }
74 else if (!ft_strncmp(token.start, ">>", 2))
75 {
76 token.type = TOKEN_APPEND;
77 token.length = 2;
78 write(1, "APPEND, ", 8);
79 write(1, &token.start[0], token.length);
80 write(1, "$\n", 2);
81 }
82 else if (!ft_strncmp(token.start, ">", 1))
83 {
84 write(1, "REOUT, ", 7);
85 token.type = TOKEN_REOUT;
86 write(1, &token.start[0], token.length);
87 write(1, "$\n", 2);
88 }
89 else if (token.start && *token.start && ft_isprint((int) *token.start) && !ft_isbashtoken((int) *token.start) && !ft_isspace((int) *token.start))
90 {
91 token.type = TOKEN_WORD;
92 token.length = 0;
93 while (temp[token.length] && ft_isprint((int) temp[token.length]) && !ft_isbashtoken((int) temp[token.length]) && !ft_isspace((int) temp[token.length]))
94 {
95 if(temp[token.length] == '\'' || temp[token.length] == '"')
96 token.length += return_quoted_length(&temp[token.length], temp[token.length]);
97 token.length++;
98 }
99 write(1, "WORD, ", 6);
100 write(1, &token.start[0], token.length);
101 write(1, "$\n", 2);
102 }
103 else
104 token.type = TOKEN_ERROR;
105 return (token);
106}
